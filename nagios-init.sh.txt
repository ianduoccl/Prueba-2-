#!/bin/bash
# chkconfig: 35 99 10
# description: Nagios Core Monitoring Tool

case "$1" in
    start)
        /usr/local/nagios/bin/nagios -d /usr/local/nagios/etc/nagios.cfg
        ;;
    stop)
        killall nagios
        ;;
    restart)
        $0 stop
        $0 start
        ;;
    *)
        echo "Usage: $0 {start|stop|restart}"
        exit 1
        ;;
esac
exit 0
